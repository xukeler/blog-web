# 运行入口文件(src/plateforms)

  • vue有两种版本，一种是完整版带编译器Runtime + Compiler，一种是正常开发借助webpack和vue-loader将.vue文件编译成JavaScript不带编译器的版本Runtime+Only，体积相对于完整版更小更轻量化。
  • 为什么会有两种版本，主要是使用的方式不一样，
    • 在new Vue({})传入的是template模板字符串，就需要编译器将模板字符串编译成render函数，这种方式使用vue不需要打包就能用。
    • 在new Vue({})传入的render 正常用vue-cli创建的项目都是这个版本，这个不包含编译器，但需要借助webpack和vue-loader将.vue文件编译成JavaScript，这种方式就需要打包才能使用。
  • 完整版入口 src/plateforms/web/runtime/entry-runtime-with-compiler.js
  • 不带编译器的版本入口 src/plateforms/web/runtime/entry-runtime.js
// 需要编译器，不需要打包
    new Vue({
      el: "#der",
      template: `<div>{{message}}</div>`,
      data: {
        message: "Hello Vue.js!",
      },
    });
// 不需要编译器，需要打包  
new Vue({
    render: h => h(App)//App.vue
})

import {
  query,//query节点查询器，
  mustUseProp,//判断节点的那些属性可以用props绑定
  isReservedTag,//判断节点是不是html和svg节点
  isReservedAttr,//判断属性是不是style和class
  getTagNamespace,// 获取标签名称

} from 'web/util/index'

import platformComponents from './components/index'//transition 动画组件相关内容

// install platform specific utils

Vue.config.isReservedTag = isReservedTag

Vue.config.isUnknownElement = isUnknownElement

// install platform runtime directives & components

// public mount method
// 挂载$mount方法 
Vue.prototype.$mount = function (
  el?: string | Element,
  hydrating?: boolean
): Component {
  el = el && inBrowser ? query(el) : undefined

}

        `Make sure to turn on production mode when deploying for production.\n` +
        `See more tips at https://vuejs.org/guide/deployment.html`
      )
    }
  }, 0)
}

# initGlobalAPI初始化构造函数的属性方法

  • 文件位置core/global-api/index.js
  • 这个文件主要是初始化Vue构造函数上的一些属性和方法，这些属性和方法可以通过Vue直接访问，也可以用过实例化对象访问vm.constructor，添加了全局组件keep-alive
    • Vue.options存放初始化的数据，在创建Vue实例时传入的配置对象最终要与这份配置属性合并
    • Vue.config全局配置对象
    • Vue.util暴露了一些辅助方法，官方并不建议使用
    • Vue.set设置响应式数据，触发视图更新。功能同this.$set
    • Vue.delete删除数组元素、删除对象属性，触发视图更新,功能同this.$delete
    • Vue.nextTick结束此轮循环后执行回调。常用与等待dom更新、加载完毕后.功能同this.$nextTick
    • Vue.observable将数据添加为响应式方法
    • Vue.use安装插件方法
    • Vue.mixin混入功能
    • Vue.extend创建基于Vue的子类并扩展初始内容
    • Vue.component注册全局组件
    • Vue.directive注册全局指令
    • Vue.filter注册全局过滤器
    • 全局组件keep-alive
/* @flow */

import config from '../config'
import { initUse } from './use'
import { initMixin } from './mixin'
import { initExtend } from './extend'
import { initAssetRegisters } from './assets'
import { set, del } from '../observer/index'
import { ASSET_TYPES } from 'shared/constants'
import builtInComponents from '../components/index'
import { observe } from 'core/observer/index'

import {
  warn,
  extend,
  nextTick,
  mergeOptions,
  defineReactive
} from '../util/index'

export function initGlobalAPI (Vue: GlobalAPI) {
  // config
  const configDef = {}
//   设置Object.defineProperty 的get方法和set方法
  configDef.get = () => config
  if (process.env.NODE_ENV !== 'production') {
    configDef.set = () => {
      warn(
        'Do not replace the Vue.config object, set individual fields instead.'
      )
    }
  }
    // Vue.config 全局配置对象
  Object.defineProperty(Vue, 'config', configDef)

  // exposed util methods.
  // NOTE: these are not considered part of the public API - avoid relying on
  // them unless you are aware of the risk.
//   扩展util方法，方便后面可以根据vm.constructor.util获取调用
  Vue.util = {
    warn,
    extend,
    mergeOptions,
    defineReactive
  }
// 添加全局的set($set),delete($delete)，nextTick($nextTick)的方法,括号是功能相同的api
  Vue.set = set
  Vue.delete = del
  Vue.nextTick = nextTick

  // 2.6 explicit observable API
//   添加全局的observable方法  其实就是将observe方法放到Vue上
  Vue.observable = <T>(obj: T): T => {
    observe(obj)
    return obj
  }

  Vue.options = Object.create(null) //初始化options 这个options接下来会用到
//   ASSET_TYPES就是一个数组[  'component','directive','filter'],包含三个元素
// 初始化components、directives、filter
  ASSET_TYPES.forEach(type => {
    Vue.options[type + 's'] = Object.create(null)
  })

  // this is used to identify the "base" constructor to extend all plain-object
  // components with in Weex's multi-instance scenarios.
//   用来识别构造函数来拓展普通对象
  Vue.options._base = Vue
// 添加全局组件keep-alive
  extend(Vue.options.components, builtInComponents)

  initUse(Vue)//初始化Vue.use方法
  initMixin(Vue)//初始化Vue.mixin方法
  initExtend(Vue)//初始化Vue.extend方法
  initAssetRegisters(Vue)//初始化Vue.component Vue.directive, Vue.filter方法
}

# Vue.use()注册插件

  • 文件位置core/global-api/use.js
  • Vue.use是构造函数上的方法，接受一个参数，这个参数可以是对象或者是函数，如果是对象就调用对象上的install方法，如果是函数则这个函数就会当作install方法，直接调用函数，将Vue构造函数传入。
  • 多次调用Vue.use插件也只会安装一次

TIP

# 为什么有的插件需要使用Vue.use,有的不需要直接引入使用?例如axios就不需要Vue.use

原因就是有的插件依赖于Vue的构造函数来实现插件的一些功能，所以使用Vue.use能获取到Vue构造函数，有的则不依赖Vue构造函数，也没有提供install方法

/* @flow */

import { toArray } from '../util/index'

export function initUse (Vue: GlobalAPI) {
  Vue.use = function (plugin: Function | Object) {
    // 判断Vue上是否存在_installedPlugins属性，没有添加一个赋值为空数组，用来记录已经安装的插件
    const installedPlugins = (this._installedPlugins || (this._installedPlugins = []))
    // 检测插件是否已经安装，避免重复安装
    if (installedPlugins.indexOf(plugin) > -1) {
      return this
    }

    // additional parameters
    // 把参数列表转换为数组
    const args = toArray(arguments, 1)
    // 把this也就是Vue构造函数添加到数组第一个位置
    args.unshift(this)
    // 判断传入的参数是个对象且包含install方法，调用对象上的install方法
    if (typeof plugin.install === 'function') {
      plugin.install.apply(plugin, args)
    } else if (typeof plugin === 'function') {
    // 如果传入的是一个函数，则直接调用
      plugin.apply(null, args)
    }
    // 标记当前插件已经注册，数组存在引用关系，所以也能修改Vue._installedPlugins
    installedPlugins.push(plugin)
    return this
  }
}

# Vue.extend()扩展Vue

  • 文件位置core/global-api/extend.js
  • Vue.extend()通常用于命令式插件的开发例如一些toast modal提示，例如this.$toast.message({})
/* @flow */

import { ASSET_TYPES } from 'shared/constants'
import { defineComputed, proxy } from '../instance/state'
import { extend, mergeOptions, validateComponentName } from '../util/index'

export function initExtend (Vue: GlobalAPI) {
  /**
   * Each instance constructor, including Vue, has a unique
   * cid. This enables us to create wrapped "child
   * constructors" for prototypal inheritance and cache them.
   */
//   设置当前组件的cld=0;
  Vue.cid = 0
//   递增子组件的cid
  let cid = 1

  /**
   * Class inheritance
   */
//   Vue.extend 本质就是对Vue的继承
  Vue.extend = function (extendOptions: Object): Function {
    extendOptions = extendOptions || {}//传入的组件暴露的对象包含data，methods，watch，computed等
    const Super = this//this就是Vue构造函数
    const SuperId = Super.cid 
    // 用_Ctor记录是否已经扩展过子类，如果有直接return
    const cachedCtors = extendOptions._Ctor || (extendOptions._Ctor = {})
    if (cachedCtors[SuperId]) {
      return cachedCtors[SuperId]
    }
    // 默认name 为父级的name
    const name = extendOptions.name || Super.options.name
    // 在开发环境监测name是否符合要求
    if (process.env.NODE_ENV !== 'production' && name) {
      validateComponentName(name)
    }
    // 创建子构造函数
    const Sub = function VueComponent (options) {
      this._init(options)
    }
    // 子构造函数的原型链继承
    Sub.prototype = Object.create(Super.prototype)
    // 修改constructor指向
    Sub.prototype.constructor = Sub
    // 自增cid 和父类区分开
    Sub.cid = cid++
    // 合并父类的options和传入的options
    Sub.options = mergeOptions(
      Super.options,
      extendOptions
    )
    // 定义supper属性
    Sub['super'] = Super

    // For props and computed properties, we define the proxy getters on
    // the Vue instances at extension time, on the extended prototype. This
    // avoids Object.defineProperty calls for each instance created.
    // 初始化子类的prop
    if (Sub.options.props) {
      initProps(Sub)
    }
    // 初始化计算属性
    if (Sub.options.computed) {
      initComputed(Sub)
    }

    // allow further extension/mixin/plugin usage
    // 属性和方法继承父级
    Sub.extend = Super.extend
    Sub.mixin = Super.mixin
    Sub.use = Super.use

    // create asset registers, so extended classes
    // can have their private assets too.
    // ASSET_TYPES=[  'component','directive','filter']
    // 继承父类的'component','directive','filter'方法
    ASSET_TYPES.forEach(function (type) {
      Sub[type] = Super[type]
    })
    // enable recursive self-lookup
    if (name) {
      Sub.options.components[name] = Sub
    }

    // keep a reference to the super options at extension time.
    // later at instantiation we can check if Super's options have
    // been updated.
    // 记录父类的options,和传入的options,在Vue实例化检测更新
    Sub.superOptions = Super.options
    Sub.extendOptions = extendOptions
    Sub.sealedOptions = extend({}, Sub.options)

    // cache constructor
    // 缓存子类的构造函数，避免再次继承
    cachedCtors[SuperId] = Sub
    return Sub
  }
}
// 把options.props设置到Comp原型上的_props 也就是Comp.prototype._props
function initProps (Comp) {
  const props = Comp.options.props
  for (const key in props) {
    proxy(Comp.prototype, `_props`, key)
  }
}
// 把子类的computed添加到构造函数的原型上，并添加响应式，
function initComputed (Comp) {
  const computed = Comp.options.computed
  for (const key in computed) {
    defineComputed(Comp.prototype, key, computed[key])
  }
}

/* @flow */

import { ASSET_TYPES } from 'shared/constants'
import { isPlainObject, validateComponentName } from '../util/index'

export function initAssetRegisters (Vue: GlobalAPI) {
  /**
   * Create asset registration methods.
   */
  // // ASSET_TYPES=[  'component','directive','filter']
  ASSET_TYPES.forEach(type => {
    // 接收两个参数
    Vue[type] = function (
      id: string,
      definition: Function | Object
    ): Function | Object | void {
      // 没有传第二个参数，就返回指定的方法
      if (!definition) {
        return this.options[type + 's'][id]
      } else {
        /* istanbul ignore if */
        // 开发环境要验证id是否满足要求
        if (process.env.NODE_ENV !== 'production' && type === 'component') {
          validateComponentName(id)
        }
        // isPlainObject检测是否是普通对象
        // 注册组件，this.options._base就是Vue构造函数
        if (type === 'component' && isPlainObject(definition)) {
          // 设置组件名称
          definition.name = definition.name || id
          //等同与 definition=Vue.extend(definition)
          definition = this.options._base.extend(definition)
        }
        // 注册指令
        if (type === 'directive' && typeof definition === 'function') {
          // 如果传入的是函数就设置bind和update钩子
          definition = { bind: definition, update: definition }
        }
        // 添加到options缓存
        this.options[type + 's'][id] = definition
        return definition
      }
    }
  })
}

/* @flow */

import config from '../config'
import { initProxy } from './proxy'
import { initState } from './state'
import { initRender } from './render'
import { initEvents } from './events'
import { mark, measure } from '../util/perf'
import { initLifecycle, callHook } from './lifecycle'
import { initProvide, initInjections } from './inject'
import { extend, mergeOptions, formatComponentName } from '../util/index'

let uid = 0//全局变量

export function initMixin (Vue: Class<Component>) {
    // 在Vue原型添加_init()方法
  Vue.prototype._init = function (options?: Object) {
    const vm: Component = this
    // a uid
    vm._uid = uid++// _uid 每个组件初始化的唯一私有标识，每次new Vue()都会递增

    let startTag, endTag
    /* istanbul ignore if */
    // 测试代码执行性能 参考window.performance
    if (process.env.NODE_ENV !== 'production' && config.performance && mark) {
      startTag = `vue-perf-start:${vm._uid}`
      endTag = `vue-perf-end:${vm._uid}`
      mark(startTag)
    }

    // a flag to avoid this being observed
    // 标识对象是不是Vue实例对象，避免对实例对象进行操作
    vm._isVue = true
    // merge options
    // 合并选项，如果是组件，调用initInternalComponent方法，如果不是组件将options的相关属性和方法合并到vm.$options
    if (options && options._isComponent) {
      // optimize internal component instantiation
      // since dynamic options merging is pretty slow, and none of the
      // internal component options needs special treatment.
      initInternalComponent(vm, options)
    } else {
      vm.$options = mergeOptions(
        resolveConstructorOptions(vm.constructor),//在init.js分析
        options || {},
        vm
      )
    }
    /* istanbul ignore else */
    // 判断是开发环境就设置vm._renderProxy,不是vm_renderProxy赋值vm
    if (process.env.NODE_ENV !== 'production') {
      initProxy(vm)
    } else {
      vm._renderProxy = vm
    }
    // expose real self
    // 定义实例私有属性vm._self = vm ，用于访问实例的数据和方法
    vm._self = vm
    initLifecycle(vm)//确认组件的父子关系，初始化实例属性$parent,生命周期
    initEvents(vm)//将父组件的自定义事件传递给子组件；初始化实例内部属性_events（事件中心）、_hasHookEvent；
    initRender(vm)//提供将render函数转为vnode的方法；初始化实例属性$slots、$scopedSlots；定义实例方法$createElement；定义响应式属性：$attrs、$listeners；
    callHook(vm, 'beforeCreate')//执行组件的beforeCreate生命周期
    initInjections(vm) // 获取inject定义的各属性值并增加对该属性的监听。
    initState(vm)//对prop，method，data，computed，watch的初始化。
    initProvide(vm) // 设置 vm._provided值，以便子组件逐级查找。
    callHook(vm, 'created')//执行组件的created生命周期

    /* istanbul ignore if */
    // 测试代码执行性能 参考window.performance
    if (process.env.NODE_ENV !== 'production' && config.performance && mark) {
      vm._name = formatComponentName(vm, false)
      mark(endTag)
      measure(`vue ${vm._name} init`, startTag, endTag)
    }

    if (vm.$options.el) {
        // 挂载
      vm.$mount(vm.$options.el)
    }
  }
}

export function initInternalComponent (vm: Component, options: InternalComponentOptions) {
  const opts = vm.$options = Object.create(vm.constructor.options)
  // doing this because it's faster than dynamic enumeration.
  const parentVnode = options._parentVnode
  opts.parent = options.parent
  opts._parentVnode = parentVnode

  const vnodeComponentOptions = parentVnode.componentOptions
  opts.propsData = vnodeComponentOptions.propsData
  opts._parentListeners = vnodeComponentOptions.listeners
  opts._renderChildren = vnodeComponentOptions.children
  opts._componentTag = vnodeComponentOptions.tag

  if (options.render) {
    opts.render = options.render
    opts.staticRenderFns = options.staticRenderFns
  }
}

export function resolveConstructorOptions (Ctor: Class<Component>) {
  let options = Ctor.options
  if (Ctor.super) {
    const superOptions = resolveConstructorOptions(Ctor.super)
    const cachedSuperOptions = Ctor.superOptions
    if (superOptions !== cachedSuperOptions) {
      // super option changed,
      // need to resolve new options.
      Ctor.superOptions = superOptions
      // check if there are any late-modified/attached options (#4976)
      const modifiedOptions = resolveModifiedOptions(Ctor)
      // update base extend options
      if (modifiedOptions) {
        extend(Ctor.extendOptions, modifiedOptions)
      }
      options = Ctor.options = mergeOptions(superOptions, Ctor.extendOptions)
      if (options.name) {
        options.components[options.name] = Ctor
      }
    }
  }
  return options
}

function resolveModifiedOptions (Ctor: Class<Component>): ?Object {
  let modified
  const latest = Ctor.options
  const sealed = Ctor.sealedOptions
  for (const key in latest) {
    if (latest[key] !== sealed[key]) {
      if (!modified) modified = {}
      modified[key] = latest[key]
    }
  }
  return modified
}
