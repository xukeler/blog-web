    # Vue原理解析-数据监听

    • vue响应式数据变化原理是采用数据劫持结合发布者-订阅者模式的方式，通过Object.defineProperty()来劫持对象各个属性的setter，getter，在数据变动时发布消息给订阅者，触发相应的监听回调。

    # 数组监听

    • 数组的监听和对象的监听有些不一样，因为Object.defineProperty对数组的支持不够友好，所以得换个方法。
    • Vue对数组的监听，采用拦截器覆盖数组原型的方法追踪变化，具体思路如下。
      • 创建一个拦截器对能改变数组方法'push', "pop", "shift", "unshift", "splice", "sort", "reverse"进行拦截。用Object.defineProperty对原型对象上操作数组的方法value赋值重写。
      • 为了防止污染全局的Array.prototype，将需要监听数据变化的数组的__proto__覆盖原型的方法，即__proto__指向以Array.prototype新建的对象Object.create(Array.prototype)，这样数组调用'push等方法就会沿着__proto__查找，而不会影响全局。
      • 数组变化，也是要收集依赖watcher，和对象一样也是在getter收集，数组变化要在拦截器中向watcher发送通知，所以拦截器中必须能够访问到dep。所以存储依赖不能存到defineReactive，拦截器能拦截到数组变化，可以把依赖watcher存在数组的属性中。需要遍历数据把每个数组都添加这个存储依赖watcher的属性，所以最好是在Observer添加属性存储。
      • 除了监听数组已有的数据，还有监听变动的数据，例如push unshift splice,所以要在拦截器中判断，如果触发的是push unshift splice，对添加的值进行监听。

    注意

    缺点：数组的监听返回的新值和旧值相同，通过索引修改值是监听不到的。

    # observe 获取Observer实例

    • 这个方法就是判断是否存在__ob__这个属性，以及这个属性是否在Observer的原型链上，如果有就代表当前的value以及被转换成了响应式数据，且可以通过__ob__访问到Observer实例对象。
        function isObject(obj) {  //判断是不是数组和对象
        return obj !== null && typeof obj === 'object'
    }
    
    function hasOwn(obj, key) {  
        const hasOwnProperty = Object.prototype.hasOwnProperty //会返回一个布尔值，判断对象自身属性中是否具有指定的属性
        return hasOwnProperty.call(obj, key)
    }
    
    function observe(value) {
        if (!isObject(value)) { //如果value不是数组或对象return
            return
        }
        if (value.__ob__ instanceof Observer && hasOwn(value, "__ob__")) {  // instanceof 运算符用于检测构造函数的 prototype 属性是否出现在某个实例对象的原型链上。
            return value.__ob__
        }
        return new Observer(value)
    }

    # def

    • 一个工具类方法，在vue源码可以找到，功能就是在obj添加属性，设置不可枚举属性值。
    
    function def(obj, key, val, enumerable) {  //在obj添加属性，设置属性值
        Object.defineProperty(obj, key, {
            value: val,
            enumerable: !!enumerable,
            writable: true,
            configurable: true
        })
    }

    # defineReactive 定义响应式数据

    • 方法结合了对象监听
    function defineReactive(obj, attr, val) {
        // 判断传参是否有两个，如果只有两个，val就等于对象属性的原始值。
        if (arguments.length === 2) {
            val = obj[attr]
        }
        // 递归子属性
        const arrOb = observe(val)  //获取Observer实例 dep存在
        let dep = new Dep()
        // 通过闭包的方式一直访问设置val
        Object.defineProperty(obj, attr, {
            enumerable: true,//enumerable 描述属性是否会出现在for in 或者 Object.keys()的遍历中
            configurable: true,//configrable 描述属性是否配置，以及可否删除
            get() {
                dep.depend()  //添加watcher
                if (arrOb) {
                    arrOb.dep.depend()
                }
                return val
            },
            set(newVal) {
                if (newVal === val) return
                val = newVal;
                dep.notify()  //通知更新
            }
        })
    }

    # 数组拦截器

        const arrayPrototype = Array.prototype;
    const arrayMethods = Object.create(arrayPrototype)//创建一个新对象，使用现有的对象来提供新创建的对象的__proto__;
    const arr = ['push', "pop", "shift", "unshift", "splice", "sort", "reverse"]//这几种方法能改变数组
    arr.forEach((item) => {
        const methods = arrayPrototype[item]; //获取Array的数组操作方法的值即push:function{}
        Object.defineProperty(arrayMethods, item, {  //监听新对象arrayMethods上面的数组方法，当使用arrayMethods上面的数组方法，就会触发value，实际触发的是value的值函数
            value (...args)  {
                const result = methods.apply(this, args); 
                const ob = this.__ob__; //获取Oberver实例对象
                let inserted
                switch (item) {  
                    case "push":
                    case "unshift":
                        inserted = args;
                        break;
                    case "splice":
                        inserted = args.slice(2)
                        break;
                }
                if (inserted) {  //对新增的数据添加监听
                    ob.observeArray(inserted)
                }
                ob.dep.notify() //通知watcher
                return result  //实现数组的功能还是没变，就能在函数内部做一些其它的事
            }
        })
    })

    # Observer类 将数据内属性全部加上getter/setter

    • 结合了对象监听
    • 判断浏览器支不支持__proto__,，不支持就会把原型上的所有属性方法加到数组上。
    const hasProto = "__proto__" in {}; //判断浏览器支不支持__proto__
    const attrKeys = Object.getOwnPropertyNames(arrayMethods)//获取arrayMethods 上面的所有属性包括不可枚举的
    class Observer {
        constructor(value) { //传递的引用类型的值，数组或对象
            this.value = value;
            this.dep = new Dep();
            def(value, "__ob__", this) //设置Observer实例在value上
            if (Array.isArray(value)) { //如果不是数组调用walk
                if (hasProto) {
                    value.__proto__ = arrayMethods//value的原型对象指向 arrayMethods，调用数组操作方法，就会调的是arrayMethods上的方法，从而实现了原型覆盖,防止污染全局Array
                    return
                }
                for (let i = 0; i < attrKeys.length; i++) {  //浏览器不支持__proto__,就把所有arrayMethods的属性添加到value上面，即数组上面
                    def(value, attrKeys[i], arrayMethods[attrKeys[i]])
                }
                this.observeArray(value)
            } else {//如果不是数组调用walk
                this.walk(value)
            }
        }
        walk(value) {
            const keys = Object.keys(value);//遍历对象，给对象的每个属性都添加getter/setter
            for (let i = 0; i < keys.length; i++) {
                defineReactive(value, keys[i], value[keys[i]])
            }
        }
        observeArray(value) { //对于数组新加的值添加监听。
            for (let i = 0; i < value.length; i++) {
                observe(this.value[i])
            }
        }
    }

    # Dep类 存储依赖watcher

    • 没有改动
    class Dep {
        //static target=null 用来存在watcher实例化对象,在源码中是这样存储的，为了方便理解，我们用window.target
        constructor() {
            this.subs = []
        }
        addSub(sub) { //将watcher 都存储到subs数组里面。
            this.subs.push(sub) //sub就是收集到的依赖watcher
        }
        removeSub(sub) {
            if (!this.subs.length) return
            let index = this.subs.indexOf(sub)//获取依赖在数组的索引
            if (index > -1) {
                this.subs.splice(index, 1)
            }
        }
        depend() {
            if (window.target) {
                this.addSub(window.target)
            }
        }
        notify() { //通知数据发生变化，执行视图更新
            const subs = this.subs.slice()//浅拷贝
            for (let i = 0; i < subs.length; i++) {
                subs[i].update() //update是依赖watcher下的一个方法，触发数据更新后的回调
            }
        }
    }

    # Watcher类

    • expOrFun什么时候会是字符串，例如我们正常使用的时候，watch: { x: fn }, Vue内部会将 x 这个key 转化为字符串。什么时候会是函数 Vue 初始化时，就是传入的渲染函数new Watcher(vm, updateComponent, ...),updateComponent就是渲染函数。
    • 没有改动
    class Watcher {
        constructor(vm, expOrFun, callback) {//vm是vue的实例，简写vm。expOrFun是字符串或函数。callback回调函数
            this.vm = vm;
            if (typeof expOrFun === "function") {
                this.getter = expOrFun;
            } else {
                this.getter = parsePath(expOrFun);
            }
            this.value = this.get();
            this.callback = callback;
        }
        get() {
            window.target = this //临时存储当前watcher实例
            let value = this.getter.call(this.vm,this.vm);
            window.target = null;
            return value
        }
        update() {
            const oldValue = this.value;
            this.value = this.get();
            this.callback.call(this.vm, this.value, oldValue)//数据改变，执行回调，this志向vm实例
        }
    }
    // 查找监听属性值
    function parsePath(path) {
        const patt = /[^\w.$]/;
        if (patt.test(path)) {
            return
        }
        const segments = path.split(".")
        return function (obj) {
            for (let i = 0; i < segments.length; i++) {
                if (!obj) return
                obj = obj[segments[i]]
            }
            return obj
        }
    }
