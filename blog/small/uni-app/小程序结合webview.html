# 小程序结合webview

  • 最近在开发移动端的需求发现了一个问题，一个需求要修改的模块，不仅要在小程序里面开发一套，还要在移动h5开发一套，这样就导致了开发资源的浪费，开发工作有50%是重复的，而且两套代码也不利于后期的维护，虽然业务逻辑一样，但是h5和小程序的代码特性以及平台特性会导致实现逻辑的代码有区别，从而影响开发质量。
  • 解决这个问题就是在小程序里面使用webview嵌套h5，但是不能全部嵌套，一些需要原生功能的页面和一些需要seo的页面不要嵌套。
  • 解决嵌套webview登录状态传递问题，这个需要后端的支持
    • 小程序登录状态传递到h5，如果h5是jwt登录，就将token通过webview url传递到h5，如果是session登录就需要后端提供解决方案，把toekn传递给后端，后端再种cookie。
    • h5登录状态传递到小程序，原理类似，
  • 在h5里需要处理跳小程序页面，和h5页面，最好是做一个封装。

WARNING

h5里面注意要处理接口错误的重定向，或者是业务里面的重定向，尽量避免跳出当前项目。

# 基于nuxt的跳转封装

  • ua.js参考常用工具类方法总结。
import userAgent from "./ua.js";
import webviewRoutes from "./configuration/webviewRoute.js";
export function getQueryString(obj) {
  let arr = [];
  Object.keys(obj).forEach(key => {
    arr.push(`${key}=${obj[key]}`);
  });
  if (arr.length) {
    return `?${arr.join("&")}`;
  }
  return "";
}
export function miniRoute(route, url) {
  let {
    isQQMiniApp,
    isBaiduMiniApp,
    isWeixinMiniApp,
    isTouTiaoMiniApp
  } = userAgent(navigator.userAgent);
  if (isQQMiniApp || isWeixinMiniApp) {
    wx && wx.miniProgram[route]({ url });
    return;
  }
  if (isBaiduMiniApp) {
    swan && swan.webView[route]({ url });
    return;
  }
  if (isTouTiaoMiniApp) {
    tt && tt.miniProgram[route]({ url: url });
  }
}

export function uniteRoute(data = {}) {
  const { key, params, query, route = "push", customLink = "" } = data;
  const currentVal = webviewRoutes[key];
  if (currentVal) {
    const { isMiniApp } = userAgent(navigator.userAgent);
    if (params) {
      currentVal.mLink += params;
    }
    if (query) {
      currentVal.mLink += getQueryString(query);
      currentVal.miniLink += getQueryString(query);
    }
    if (customLink) {
      currentVal.mLink = customLink;
      currentVal.miniLink = customLink;
    }
    switch (route) {
      case "push":
        if (isMiniApp) {
          miniRoute("navigateTo", currentVal.miniLink);
          return;
        }
        window.$nuxt.$router.push(currentVal.mLink);
        break;
      case "replace":
        if (isMiniApp) {
          miniRoute("reLaunch", currentVal.miniLink);
          return;
        }
        window.$nuxt.$router.replace(currentVal.mLink);
        break;
      case "href":
        if (isMiniApp) {
          miniRoute("navigateTo", currentVal.miniLink);
          return;
        }
        window.location.href = currentVal.mLink;
        break;
    }
  }
}
