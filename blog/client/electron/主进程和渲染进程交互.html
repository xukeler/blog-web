    # 主进程和渲染进程交互

    • 主进程是处理整个electron的生命周期，包括各种原生的各种状态，同时也要和渲染进程交互，主进程可以调用原生的模块，包括调用node.js。
    • 通过new BrowserWindow来创建一个窗口，渲染html界面，也可以load一个远程的url例如https://www.baidu.com。
    const winURL = process.env.NODE_ENV === 'development'
  ? `http://localhost:9080` // 开发模式的话走webpack-dev-server的url
  : `file://${__dirname}/index.html`

function createWindow () {
  /**
   * Initial window options
   */

    fullscreen: false,
    show: false ,
    useContentSize: false,
    width:900,
    height:600,
    center:true,
    minWidth:900,
    minHeight:600,
  })
  // mainWindow.maximize()
  mainWindow.loadURL(winURL)
  // mainWindow.webContents.openDevTools()
  // mainWindow.setMenu()
  mainWindow.once('ready-to-show', () => {
    
    mainWindow.show()
  })
  mainWindow.webContents.on('did-finish-load', () => {
    mainWindow.webContents.send('img-files', process.argv)
  })
  mainWindow.on('closed', () => {
    mainWindow = null
  })
  mainWindow.on('close', () => {
    app.exit()
  })
}

    # 交互

    • 一般来说主进程不能主动向渲染进程发送消息，渲染进程可以主动向主进程发消息，主进程响应渲染进程发的消息，但是也有一种方法可以让主进程主动向渲染进程发送消息。
    • 主进程主动发消息通过webContents.send方法发送，必须在did-finish-load加载完成之后发送，在渲染进程引入ipcRenderer，约定的事件名称要一致。
    • 渲染进程主动发消息通过ipcRenderer的send方法，主进程通过ipcMain来响应渲染进程发的消息，事件的名称要一致。
    //主进程
  mainWindow = new BrowserWindow();
  //主动发消息
  mainWindow.webContents.on('did-finish-load', () => {
    mainWindow.webContents.send('img-files', process.argv)
  
//渲染进程
 const {ipcRenderer} = require('electron')
 //接收到消息

  })


//渲染进程主动发消息
ipcRenderer.on('asynchronous-reply', (event, arg) => {
  console.log(arg) // prints "pong" //响应主进程发的消息
})
ipcRenderer.send('asynchronous-message', 'ping')//发消息

//主进程

const {ipcMain} = require('electron')
ipcMain.on('asynchronous-message', (event, arg) => {
  console.log(arg)  // prints "ping"
  event.sender.send('asynchronous-reply', 'pong')  //接收到渲染进程的消息，再发送给渲染进程
})

ipcMain.on('synchronous-message', (event, arg) => { //两种发送的形式。
  console.log(arg)  // prints "ping"
  event.returnValue = 'pong'
})
